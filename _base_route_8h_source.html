217 template <typename SettingsType>
218 bool BaseRoute_<SettingsType>::canHandleRequest(const Poco::Net::HTTPServerRequest& request,
219  bool isSecurePort) const
220 {
221  // If this isn't a secure port and we require that, reject it.
222  if (_settings.requireSecurePort() && !isSecurePort)
223  {
224  return false;
225  }
226 
227  // Check the request method.
228  const BaseRouteSettings::HTTPMethodSet& validHTTPMethods = _settings.getValidHTTPMethods();
229 
230  // If validHTTPMethods are defined, then the request must match.
231  if (!validHTTPMethods.empty() && validHTTPMethods.find(request.getMethod()) == validHTTPMethods.end())
232  {
233  return false;
234  }
235 
236  // Check the request content type.
237  const BaseRouteSettings::MediaTypeSet& validContentTypes = _settings.getValidContentTypes();
238 
239  if (!validContentTypes.empty())
240  {
241  BaseRouteSettings::MediaTypeSet::iterator iter = validContentTypes.begin();
242 
243  const std::string& contentType = request.getContentType();
244 
245  bool foundMatch = false;
246 
247  while (iter != validContentTypes.end())
248  {
249  Poco::Net::MediaType type(*iter);
250 
251  if (type.matchesRange(contentType))
252  {
253  foundMatch = true;
254  break;
255  }
256 
257  ++iter;
258  }
259 
260  if (!foundMatch)
261  {
262  return false;
263  }
264  }
265 
266  // require a valid path
267  std::string path = "/";
268 
269  try
270  {
271  path = Poco::URI(request.getURI()).getPath();
272  }
273  catch (const Poco::SyntaxException& exc)
274  {
275  ofLogError("BaseRoute::canHandleRequest") << exc.displayText();
276  return false;
277  }
278 
279  if (path.empty())
280  {
281  path = "/";
282  }
283 
284  try
285  {
286  // \TODO cache this regex
287  return Poco::RegularExpression(routePathPattern()).match(path);
288  }
289  catch (const Poco::RegularExpressionException& exc)
290  {
291  ofLogError("BaseRoute::canHandleRequest") << exc.displayText();
292  return false;
293  }
294 
295 }

308 template <typename SettingsType>
309 void BaseRoute_<SettingsType>::handleRequest(Poco::Net::HTTPServerRequest& request,
310  Poco::Net::HTTPServerResponse& response)
311 {
312 // std::cout << "BaseRoute_<SettingsType>::handleRequest" << std::endl;
313  // We assert that the server must be set on a route.
314  poco_assert(_server);
315 
316  // All requests pass through the server's handleRequest method first.
317  // The server broadcasts the request / response to any listeners.
318  // Generally these listeners should only modify headers, not send a response.
319  AbstractSession& session = _server->sessionStore().getSession(request, response);
320 
321  ServerEventArgs evt(request, response, session);
322 
323  _server->onHTTPServerEvent(this, evt);
324 
325  // If the response was sent from the server or its delegates, we finish.
326  if (response.sent())
327  {
328  return;
329  }
330  else
331  {
332  // Otherwise, we call the route's sendResponse method.
333  // Eventually we may end up back at this base class' sendResponse
334  // method.
335  handleRequest(evt);
336  }
337 
338  // A failsafe.
339  if (!response.sent())
340  {
341  BaseRoute_<SettingsType>::handleRequest(evt);
342  }
343 }

346 template <typename SettingsType>
347 void BaseRoute_<SettingsType>::handleRequest(ServerEventArgs& evt)
348 {
349 // std::cout << "BaseRoute_<SettingsType>::handleRequest(ServerEventArgs& evt" << std::endl;
350 
351  if (evt.response().sent())
352  {
353  return;
354  }
355 
356  // We gave the handlers every opportunity to send a response.
357  // Now we must conclude that there is a server error.
358  try
359  {
360  // If we got this far and our status is still marked as 200,
361  // that constitutes a server error.
362  if (Poco::Net::HTTPResponse::HTTP_OK == evt.response().getStatus())
363  {
364  evt.response().setStatusAndReason(Poco::Net::HTTPResponse::HTTP_INTERNAL_SERVER_ERROR,
365  "No handlers for route.");
366  }
367 
368  evt.response().setChunkedTransferEncoding(true);
369  evt.response().setContentType("text/html");
370 
371  std::stringstream ss;
372  ss << evt.response().getStatus() << " - " << evt.response().getReason();
373 
374  std::ostream& ostr = evt.response().send(); // get output stream
375  ostr << "<!DOCTYPE html><html><head><meta charset=\"utf-8\"/><title>";
376  ostr << ss.str();
377  ostr << "</title></head><body><h1>";
378  ostr << ss.str();
379  ostr << "</h1></body></html>";
380  ostr.flush();
381  }
382  catch (const Poco::Exception& exc)
383  {
384  ofLogError("BaseRoute::handleRequest") << "Exception: " << exc.code() << " " << exc.displayText();
385  }
386  catch (const std::exception& exc)
387  {
388  ofLogError("BaseRoute::handleRequest") << "exception: " << exc.what();
389  }
390  catch ( ... )
391  {
392  ofLogError("BaseRoute::handleRequest") << "... Unknown exception.";
393  }
394 }
