188  void start()
189  {
190  if (isRunning())
191  {
192  ofLogWarning("BaseServer_::start") << "Server is already running. Call stop() to stop.";
193  return;
194  }
195 
196  Poco::Net::ServerSocket socket;
197 
198  if (_settings.useSSL())
199  {
200  // we use the default thread pool
201  try
202  {
203  socket = Poco::Net::SecureServerSocket(_settings.getPort(),
204  DEFAULT_BACKLOG,
205  ofSSLManager::getDefaultServerContext());
206 
207  }
208  catch (const Poco::Exception& exc)
209  {
210  ofLogError("BaseServer_::start") << "Poco::Exception: " << exc.name() << " " << exc.what() << " " << exc.message();
211  }
212  }
213  else
214  {
215  socket = Poco::Net::ServerSocket(_settings.getPort());
216  }
217 
218 
219  try
220  {
221  _server = std::make_shared<Poco::Net::HTTPServer>(new BaseServerHandle(*this),
222  getThreadPool(),
223  socket,
224  _toPoco(_settings));
225 
226  _isSecurePort = socket.secure();
227 
228 #if defined(TARGET_OSX)
229  // essential on a mac! fixed in 1.4.6p2+ / 1.5.2+
230  socket.setOption(SOL_SOCKET, SO_NOSIGPIPE, 1); // ignore SIGPIPE
231 #endif
232 
233  ofLogNotice("BaseServer_::start()") << "Starting server on port: " << _settings.getPort();
234 
235  // start the http server
236  _server->start();
237  }
238  catch (const Poco::Net::InvalidSocketException& exc)
239  {
240  ofLogError("BaseServer_::start()") << exc.displayText();
241  }
242  }

245  void stop()
246  {
247  if (!isRunning())
248  {
249  ofLogVerbose("BaseServer_::stop") << "Server is not running. Call start() to start.";
250  return;
251  }
252 
253  Routes::const_reverse_iterator iter = _routes.rbegin();
254 
255  while (iter != _routes.rend())
256  {
257  (*iter)->stop();
258  ++iter;
259  }
260 
261  // Stop our default route.
262  _defaultRoute.stop();
263 
264  #if POCO_VERSION > 0x01040600
265  _server->stopAll(true);
266  #else
267  _server->stop();
268  #endif
269 
270  ofLogVerbose("BaseServer_::stop") << "getThreadPool().capacity() = " << getThreadPool().capacity();
271  ofLogVerbose("BaseServer_::stop") << "getThreadPool().getStackSize() = " << getThreadPool().getStackSize();
272  ofLogVerbose("BaseServer_::stop") << "getThreadPool().used() = " << getThreadPool().used();
273  ofLogVerbose("BaseServer_::stop") << "getThreadPool().allocated() = " << getThreadPool().allocated();
274  ofLogVerbose("BaseServer_::stop") << "getThreadPool().available() = " << getThreadPool().available();
275 
276  // Wait for all threads in the thread pool.
277  // Particularly troubling if we are sharing
278  // this pool with other non-server-based-processes.
279  // getThreadPool().joinAll();
280 
281  getThreadPool().stopAll();
282 
283  _server.reset();
284 
285  _isSecurePort = false;
286  }

338  Poco::Net::HTTPRequestHandler* createRequestHandler(const Poco::Net::HTTPServerRequest& request)
339  {
340  if (acceptConnection(request))
341  {
342  // We start with the last factory that was added.
343  // Thus, factories with overlapping routes should be
344  // carefully ordered.
345  auto iter = _routes.rbegin();
346 
347  while (iter != _routes.rend())
348  {
349  if ((*iter)->canHandleRequest(request, _isSecurePort))
350  {
351  return (*iter)->createRequestHandler(request);
352  }
353 
354  ++iter;
355  }
356  }
357 
358  return _defaultRoute.createRequestHandler(request);
359  }

407  bool acceptConnection(const Poco::Net::HTTPServerRequest& request)
408  {
409 #if POCO_VERSION > 0x01050000
410  const Poco::Net::IPAddress& host = request.clientAddress().host();
411 
412  // If a whitelist is defined then you _must_ be on the whitelist.
413  const Net::IPAddressRange::List& whitelist = _settings.getWhitelist();
414 
415  if (!whitelist.empty())
416  {
417  bool isWhitelisted = false;
418 
419  Net::IPAddressRange::List::const_iterator iter = whitelist.begin();
420 
421  while (iter != whitelist.end())
422  {
423  if (iter->contains(host))
424  {
425  return true;
426  }
427 
428  ++iter;
429  }
430 
431  if (!isWhitelisted)
432  {
433  return false;
434  }
435  }
436 
437  // If a blacklist is defined then you _must not_ be on the blacklist.
438  const Net::IPAddressRange::List& blacklist = _settings.getBlacklist();
439 
440  if (!blacklist.empty())
441  {
442  Net::IPAddressRange::List::const_iterator iter = blacklist.begin();
443 
444  while (iter != blacklist.end())
445  {
446  if (iter->contains(request.clientAddress().host()))
447  {
448  return false;
449  }
450 
451  ++iter;
452  }
453  }
454 #endif
455 
456  return true;
457  }

464  Poco::Net::HTTPServerParams::Ptr _toPoco(const HTTPServerParams& params)
465  {
466  // hack!
467  // all of these params are an attempt to make the server shut down VERY quickly.
468  Poco::Net::HTTPServerParams::Ptr serverParams = new Poco::Net::HTTPServerParams();
469 
470  serverParams->setMaxQueued(params.getMaxQueued());
471 
472  // TODO: currently a bug (?) in void TCPServerParams::setMaxThreads(int count).
473  // Poco::Net::HTTPServerParams::setMaxThreads() should be able to handle
474  // a value of 0 (according to the documentation), but it is currently asserting
475  // that the value must be > 0. This is fixed in Poco 1.6+.
476  if (params.getMaxThreads() <= 0)
477  {
478  serverParams->setMaxThreads(getThreadPool().capacity());
479  }
480  else
481  {
482  serverParams->setMaxThreads(params.getMaxThreads());
483  }
484 
485  serverParams->setKeepAlive(params.getKeepAlive());
486  serverParams->setMaxKeepAliveRequests(params.getMaxKeepAliveRequests());
487  serverParams->setKeepAliveTimeout(params.getKeepAliveTimeout());
488  serverParams->setServerName(params.getServerName());
489  serverParams->setTimeout(params.getTimeout());
490  serverParams->setThreadIdleTime(params.getThreadIdleTime());
491  serverParams->setThreadPriority(params.getThreadPriority());
492  serverParams->setSoftwareVersion(params.getSoftwareVersion());
493 
494  return serverParams;
